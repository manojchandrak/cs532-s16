		function fixPhotoViewPos(){
				var option = 0;

				var winHeight = $(window).height() - 40;
				var winWidth = $(window).width();

				var ori_picHeight = $("#reporter_detail img").attr("max_height");
				var ori_picWidth = $("#reporter_detail img").attr("max_width");

				var title_height = $("#photoTextBox").height() + 40;
				var title_width = $("#photoTextBox").width();

				//var ad_height = $("#photoAds").height();
				//var ad_width = $("#photoAds").width();

				var ad_height = 0;
				var ad_width = 0;


				/* Photo Size Setting */
				if(ori_picHeight > winHeight - (title_height + ad_height + 66)){
					option = option + 1;
				}

				if(ori_picWidth > winWidth){
					option = option + 2;
				}
				//console.log(option);
				switch(option){// Image Setting
					case 1: // 세로만 긴 경우
						//가로 길이 계산
						//$("#photoViewer").css("width",winWidth + "px");
						//$("#photoViewer").css("height",winHeight + "px");

	function shareKakaoStory(){

	        // 로그인 성공시, API를 호출합니다.

	        Kakao.API.request( {
	          url : '/v1/api/story/linkinfo',
	          data : {
	            url : 'http://www.metroseoul.co.kr/news/newsview?newscd=2016021100022'
	          }
	        }).then(function(res) {// 이전 API 호출이 성공한 경우 다음 API를 호출합니다.
	          return Kakao.API.request( {
	            url : '/v1/api/story/post/link',
	            data : {
	              link_info : res
	            }
	          });
	        }).then(function(res) {
	          return Kakao.API.request( {
	            url : '/v1/api/story/mystory',
	            data : { id : res.id }
	          });
