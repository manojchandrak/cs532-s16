    function getEndHashtags(str) {
        var m = str.replace(endLinkMatcher, '').match(/(\s*#[a-z0-9_]+\s*)+$/ig);

        return m ? m[0].replace(/\s*#/, '').split(/\s*#/g) : [];
    }

    function deconstructTweet(tweet, defaultLink) {
        var data = {};
        var replaced = tweet.replace(manualLinkMatcher, defaultLink);

        data.text = replaced.replace(endHastagMatcher, '').replace(endLinkMatcher, '').replace(endHastagMatcher, '').strip();
        data.hashtags = getEndHashtags(replaced).join(',');
        var middleLinkMatch = replaced.replace(endHastagMatcher, '').replace(endLinkMatcher, '').match(linkMatcher);

        if(!tweet.match(manualLinkMatcher) && middleLinkMatch) {
            data.link = 'N'+Math.random(); // No link.
        } else {
            var endLinkMatch = replaced.match(endLinkMatcher);
            if(endLinkMatch) {
                data.link = endLinkMatch[0];
            } else {
                data.link = defaultLink;
            }
        }

        return data;
