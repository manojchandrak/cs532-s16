        <strong>Jakarta</strong> - USD 936 ribu atau sekitar Rp 12,5 miliar (USD 1 = Rp 13.400) adalah jumlah uang yang dihabiskan Facebook untuk membayar satuan pengamanan (satpam). Satpam yang dimaksud di sini adalah peneliti keamanan yang dipekerjakan oleh Facebook untuk menemukan celah di jejaring media sosial miliknya. <br /><br />Ada 210 peneliti keamanan yang menjadi bagian dari program 'bounty bug' Facebook. 210 peneliti itu menemukan 526 celah keamanan selama tahun 2015. Biaya yang dikeluarkan oleh Facebook ini lebih kecil dibanding pengeluaran mereka untuk urusan yang sama pada tahun 2014 dan 2013.<br /><br />Pada tahun 2014 Facebook mengeluarkan uang sebanyak USD 1,3 juta untuk 321 peneliti keamanan, dan USD 1,5 juta untuk 330 peneliti keamanan pada tahun 2013. Secara total, ada lebih dari 800 peneliti yang pernah bekerja di program tersebut sejak tahun 2011.<br /><br />Sejak program tersebut diadakan, lebih dari 2.400 celah keamanan yang dilaporkan melalui program tersebut, dikutip <strong>detikINET </strong>dari <em>Venture Beat</em>, Kamis (11/2/2016).<br /><br />Facebook menjelaskan alasan utama kenapa tahun 2015 ini semakin sedikit jumlah celah keamanan yang dilaporkan. Yaitu ada 102 laporan yang masuk ke dalam kategori berdampak besar, atau meningkat 38% dibanding 2014. Jadi, kualitas dari tiap laporan yang masuk semakin baik, dan setiap laporan makin mengarah ke logika bisnis.<!--S:AUTHOR INISIAL-->

	var selected_cat_id = '';

		//time signal newopenx finish//

		if (selected_cat_id != '') {
			$("div.tabs").find("a").each(function(index){
				$(this).removeClass("selected");
				var cat_id = '';
				if (cat_id.search($(this).attr('id')) != -1) {
					$(this).addClass("selected");
					$('div.tabs-body').hide();
					$('div.tabs-body').eq(index).show();
				}
			});
		}

		$('select.bhinneka_select').bind('change', function(){
			var cat_id = $(this).val();
			selected_cat_id = cat_id;
			var type = $(this).attr('id');

			// check level
			if ($(this).hasClass('level1')) {
				var child = '#select2_' + type;
			}
			else {
				var child = '#select3_' + type;
			}
			$.ajax({
				url: 'http://inet.detik.com/index.php?fa=detik.getbhinneka&type=category&cat_id=' + cat_id,
				dataType: 'json',
				success: function(data) {
					var options = '';
					if (data) {
						options += '<option value="">- Tipe -</option>';
						for (var i=0; i<data.length; i++) {
							options += '<option value="' + data[i]['id'] + '">' + data[i]['name'] + '</option>';
						}
					}
					$(child).html(options);
					$(child).show();
				}
			});
		});

		
		$('select.bhinneka_select_brand').change(function(){
			var cat_id = '';
			var brand_id = $(this).val();
			var type = $(this).attr('id');
			var child = '#produk_' + type;
			$.ajax({
				url: 'http://inet.detik.com/index.php?fa=detik.getbhinneka&type=product_by_brand&cat_id=' + cat_id + '&brand_id=' + brand_id,
				dataType: 'json',
				success: function(data) {
					var options = '';
					if (data) {
						for (var i=0; i<data.length; i++) {
							options += '<option value="' + data[i]['id'] + '">' + data[i]['name'] + '</option>';
						}
					}
					$(child).html(options);
				}
			});
		});

		$('select.bhinneka_pricerange').change(function(){
			selected_price = $(this).val();
		});

		$('form.bhinneka_cekharga').submit(function(){
			if (selected_cat_id == '') {
				selected_cat_id = $("div.tabs").find("a.selected").attr("id");
			}
			window.location.href = "http://inet.detik.com/listharga/" + selected_cat_id + '/' + selected_price;
			return false;
		});

		$('form#bhinneka_compare').submit(function() {
			var pid1 = $('select#produk_brand1').val();

			var pid2 = $('select#produk_brand2').val();
			var pid2_name = $('select#produk_brand2 option:selected').text();
			if (pid1=='' || pid2=='') {
				alert('Silahkan pilih produk yang ingin dibandingkan');

			else {
				window.location.href = "http://inet.detik.com/comparison/" + selected_cat_id + '/' + pid1 + '/' + pid1_name + '/' + pid2 + '/' + pid2_name;
			}
			return false;
		});
	});
