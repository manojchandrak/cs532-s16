    if(agent.search(/iPhone/) != -1 || agent.search(/iPad/) != -1 || agent.search(/iPod/) != -1 || agent.search(/Android/) != -1){
      var is_tablet = 1;
    }
    var $w = $(window),
        hh = $('#globalHeader').innerHeight(),
        fh = $('#globalFooter').innerHeight(),
        is_hdr = $('#subHeader').length,
        is_nav = $('#globalNav').length,
        is_jack = $('#jackHead').length,
        nc = is_jack?Math.abs(parseInt($('#globalFooter').css('margin-top'),10))-Math.abs(parseInt($('.contentInner').css('top'),10)):0;

    if(is_nav > 0){
        var $n = $('#globalNav'),
            nm = 0,  //globalNav margin
            nh = $('#globalNav').innerHeight() + nm * 2,
            nt = $('#globalNav').offset().top - nm;
    }
    if(is_hdr > 0) {
        var ht = $('#subHeader').offset().top,
            col2 = $('body').hasClass('col2');
    }

    var navfix = function(){
      var ws = $w.scrollTop(),
          wl = $w.scrollLeft(),
          wh = $w.innerHeight(),
          dh = $(document).innerHeight();

      if(is_hdr > 0) {
        //if(ws > ((is_jack && col2) ? 183 : 0)) {
        if(ws > ((is_jack && col2) ? 241 : 0)) {
          $("#subHeader").fadeIn(300);
          $('#subHeader').addClass('fixed');
        } else {
          $("#subHeader").fadeOut(300,function(){$('#subHeader').removeClass('fixed');});
        }
      }


      if(is_nav > 0) {
        var  nl = $('#nav').offset().left+30;

        if(nh-wh < 0) {
          if(dh-ws-fh+nc < nh){
            $n.css({'position':'absolute', 'top':'auto', 'bottom':nm + 'px', 'left':'auto'});
          }
          else if(ws > nt){
            $n.css({'position':'fixed', 'top':nm + 'px', 'bottom':'auto', 'left':(wl-nl) < 0 ? (nl-wl) + 'px' : '-' + (wl-nl) + 'px'});
          }
          else{
            $n.css({'position':'static', 'top':'auto', 'bottom':'auto', 'left':'auto'});
          }
        }else{
          if(dh-ws-wh-fh+nc < 0){
            $n.css({'position':'absolute', 'top':'auto', 'bottom':nm + 'px', 'left':'auto'});
          }
          else if(ws > nt+nh-wh){
            $n.css({'position':'fixed', 'top':'auto', 'bottom':nm + 'px', 'left':(wl-nl) < 0 ? (nl-wl) + 'px' : '-' + (wl-nl) + 'px'});
          }
          else{
            $n.css({'position':'static', 'top':'auto', 'bottom':'auto', 'left':'auto'});
          }
        }
      }
    }

    if(!is_tablet) {
      $w.on('scroll',navfix);
      $w.on('resize',navfix);
    }
    navfix();

  });
})(jQuery);
