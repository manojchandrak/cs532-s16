    function refreshHeaderMenuCart() {
        //TNDG--->Test
        var cartItemsCount = 0;

        var comicCartItemsCount = 0;
        var bookCartItemsCount = 0;

        var cookieValMusic = getCookie("MUSIC_COMM_CART");
        var arrStrMusic;
        if (cookieValMusic != null) {
            arrStrMusic = cookieValMusic.split("|");
            musicCartItemsCount = arrStrMusic.length;
            cartItemsCount += arrStrMusic.length;
        }
        var cookieValComic = getCookie("COMIC_COMM_CART");
        var arrStrComic;
        if (cookieValComic != null) {
            arrStrComic = cookieValComic.split("|");
            comicCartItemsCount = arrStrComic.length;
            cartItemsCount += arrStrComic.length;
        }

        var cookieValBook = getCookie("BOOK_COMM_CART");
        var arrStrBook;
        if (cookieValBook != null) {
            arrStrBook = cookieValBook.split("|");
            bookCartItemsCount = arrStrBook.length;
            cartItemsCount += arrStrBook.length;
        }

        if (cartItemsCount > 0) {
            if (!$("#HeaderMenuCartItemNumber").hasClass("badge"))
                $("#HeaderMenuCartItemNumber").addClass("badge");
            $("#HeaderMenuCartItemNumber").html(cartItemsCount);
        }

            if ($("#HeaderMenuCartItemNumber").hasClass("badge"))

    if (cookieVal != null && cookieVal.length > 0 && (cornerVal == null || cornerVal.length == 0)) {
        var cookieArray = cookieVal.split("|");

        for (var i = 0; i< cookieArray.length; i++) {
